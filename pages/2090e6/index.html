远程医疗协作的异地场景同步构建与虚实融合交互系统研发

# 项目描述

安徽省科技重大专项项目。项目团队利用Socket网络通讯传输方法，能够将AR眼镜采集得到的影像以及IMU数据传输至接收端。接着，通过眼镜捕获图像中的AprilTag标签推导出其所处的实时位姿信息，这使得本地患者端指导信息能够回传至AR眼镜上，为远程医疗协作提供虚实融合辅助。

# 位姿轨迹追踪

提示

需要连接外网才可查看网站中的视频内容，建议调整网络连接方式。

本项目CMakeLists.txt配置，包含库依赖、编译选项

cmake_minimum_required(VERSION 3.26)
project(ApriltagPoseEstimation)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -O3 -std=c++11 ")
set(OpenCV_DIR "/usr/local/lib/cmake/opencv4")
find_package(OpenCV REQUIRED)
set(ENV{PKG_CONFIG_PATH} "/home/alan/Software/apriltag/lib/pkgconfig:$ENV{PKG_CONFIG_PATH}")
find_package(PkgConfig)
pkg_search_module(apriltag REQUIRED apriltag)
set(apriltag_INCLUDE_DIRS "${apriltag_INCLUDE_DIRS}/apriltag")
link_directories(${apriltag_LIBDIR})

include_directories(include
        ${apriltag_INCLUDE_DIRS}
        ${OpenCV_INCLUDE_DIRS}
)

add_executable(ApriltagPoseEstimation main.cpp)
target_link_libraries(${PROJECT_NAME} ${OpenCV_LIBS} ${apriltag_LIBRARIES})
